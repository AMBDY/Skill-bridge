Proof of Payment

Upload proof so the platform can verify and hold funds in escrow.

document.getElementById('proofForm').addEventListener('submit', async (e) => { e.preventDefault(); if (!Auth.isLoggedIn()) return Toast.show('Please sign in'); const data = Object.fromEntries(new FormData(e.target)); const params = new URLSearchParams(location.search); try { await API.post('/payments', { job_id: params.get('job') || null, worker_id: params.get('worker') || null, amount: +data.amount, payment_method: params.get('method') || 'fiat_naira', proof_url: data.proof_url, proof_meta: data }); Toast.show('Proof submitted! Funds held in escrow.'); setTimeout(() => location.href = '/dashboard.html', 1200); } catch (err) { Toast.show(err.message); } });
